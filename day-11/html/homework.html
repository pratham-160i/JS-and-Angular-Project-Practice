Exercise 1: Form Validation with preventDefault

What to do:

  • Grab the form, name input, email input, and result paragraph
  • On form submit, use preventDefault() to stop page reload
  • Check if both fields are filled. Show success or error message

let loginForm = document.querySelector("#login-form");
let formResult = document.querySelector("#form-result");

loginForm.addEventListener("submit", (e) => {
  e.preventDefault();
  
  let name = document.querySelector("#form-name").value;
  let email = document.querySelector("#form-email").value;
  
  if (name === "" || email === "") {
    formResult.innerHTML = "? All fields required!";
    console.log("All fields required!");
  } else {
    formResult.innerHTML = "? Form valid, submitted successfully!";
    console.log("Form valid, submitting...");
  }
});

Exercise 2: Multiple Event Listeners

What to do:

  • Select all items with class item
  • Add a click listener that logs item text and changes background
  • Add mouseenter and mouseleave listeners for hover effects
Item 1 - Click me!
Item 2 - Click me!
Item 3 - Click me!
let items = document.querySelectorAll(".item");

items.forEach((item) => {
  item.addEventListener("click", function() {
    console.log("Clicked:", this.textContent);
    this.style.backgroundColor = "#C8E6C9";
  });
  
  item.addEventListener("mouseenter", function() {
    this.style.transform = "scale(1.05)";
    this.style.boxShadow = "0 2px 8px rgba(0,0,0,0.1)";
  });
  
  item.addEventListener("mouseleave", function() {
    this.style.transform = "scale(1)";
    this.style.boxShadow = "none";
  });
});

Exercise 3: Keyboard Events

What to do:

  • Listen for keydown events on the document
  • Check which key was pressed using e.key
  • Display different messages for Enter, Escape, and other keys

Press keys to see events (check console):

Press any key...

let keyDisplay = document.querySelector("#key-display");

document.addEventListener("keydown", (e) => {
  if (e.key === "Enter") {
    keyDisplay.textContent = "? Enter key pressed!";
    console.log("Enter pressed");
  } else if (e.key === "Escape") {
    keyDisplay.textContent = "? Escape key pressed!";
    console.log("Escape pressed");
  } else {
    keyDisplay.textContent = "?? Key pressed: " + e.key;
    console.log("Pressed:", e.key);
  }
});

Task 1 Sync vs Async Output
  • Predict the output order BEFORE running.
  • Run the code and compare the actual output.
  • Explain why Promise callbacks execute before setTimeout(). 
console.log("A");

setTimeout(() => console.log("B"), 0);

console.log("C");

Promise.resolve()
  .then(() => console.log("D"));
// Output:
//
// A
// C
// D
// B
//
// Promise callbacks are microtasks.
//
// setTimeout callbacks are macrotasks.
//
// The event loop always clears
// microtasks before macrotasks.

Task 2 Promisify a Callback API
  • Convert a callback-based function into a Promise version.
  • Chain three delayed logs. 
function delayLog(msg, ms, cb) {

  setTimeout(() => {

    console.log(msg);

    cb(null);

  }, ms);
}

function delayLogPromise(msg, ms) {

  return new Promise((resolve, reject) => {

    delayLog(msg, ms, (err) => {

      if (err) {

        reject(err);

      } else {

        resolve();
      }
    });
  });
}

delayLogPromise("1", 300)

  .then(() => delayLogPromise("2", 200))

  .then(() => delayLogPromise("3", 100))

  .catch((err) => console.log(err));

Try the steps, then compare with ../js/hands-on.js.

Task 3 Promise.all in Action
  • Create multiple async price requests.
  • Fetch all prices in parallel using Promise.all().
  • Confirm total execution time matches the slowest request. 
function fetchPrice(item, ms) {

  const prices = {

    pen: 50,
    book: 200,
    bag: 800
  };

  return new Promise((resolve) => {

    setTimeout(() => {

      resolve({
        item,
        price: prices[item]
      });

    }, ms);
  });
}

const start = Date.now();

Promise.all([

  fetchPrice("pen", 300),

  fetchPrice("book", 500),

  fetchPrice("bag", 700)

])

.then((results) => {

  const total = results.reduce(

    (sum, item) => sum + item.price,

    0
  );

  console.log("Total:", total);

  console.log(
    "Time:",
    Date.now() - start,
    "ms"
  );
});
// Promise.all runs all promises
// in parallel.
//
// Total time depends on the
// slowest promise, not the sum.

Bonus Promise.all vs Promise.allSettled
  • Compare Promise.all() and Promise.allSettled().
  • Observe how failed promises are handled. 
const p1 =
  Promise.resolve("ok1");

const p2 =
  Promise.reject(
    new Error("fail")
  );

const p3 =
  Promise.resolve("ok2");

Promise.all([p1, p2, p3])

  .then((res) => console.log(res))

  .catch((err) =>
    console.log("ALL ERROR:", err.message)
  );

Promise.allSettled([p1, p2, p3])

  .then((res) =>
    console.log(res)
  );
// Promise.all stops immediately
// if any promise fails.
//
// Promise.allSettled waits for
// every promise and returns
// both success and failure results.
//
// Use Promise.all when every result
// is required.
//
// Use Promise.allSettled when partial
// success is acceptable.

Try the steps, then compare with ../js/hands-on.js.

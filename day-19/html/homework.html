Task 1 Fetch Users in Parallel
  • Write a fetchAllUsers(ids) function.
  • Fetch users in parallel using Promise.all() and map().
  • Compare timing against a sequential version. 
function fetchUser(id) {

  return new Promise((resolve) => {

    setTimeout(() => {

      resolve({
        id,
        name: `User-${id}`
      });

    }, 500);
  });
}

async function fetchSequential(ids) {

  const start = Date.now();

  const users = [];

  for (const id of ids) {

    const user =
      await fetchUser(id);

    users.push(user);
  }

  console.log(users);

  console.log(
    "Sequential:",
    Date.now() - start,
    "ms"
  );
}

async function fetchAllUsers(ids) {

  const start = Date.now();

  const users =
    await Promise.all(

      ids.map((id) =>
        fetchUser(id)
      )
    );

  console.log(users);

  console.log(
    "Parallel:",
    Date.now() - start,
    "ms"
  );
}

fetchSequential([1, 2, 3]);

fetchAllUsers([1, 2, 3]);
// Sequential execution waits one-by-one.
//
// Promise.all runs everything together
// in parallel.

Try the steps, then compare with ../js/homework.js.

Task 2 Build withTimeout()
  • Create a withTimeout() helper.
  • Reject if the promise takes too long.
  • Use Promise.race(). 
function withTimeout(promise, ms) {

  const timeoutPromise =
    new Promise((_, reject) => {

      setTimeout(() => {

        reject("timeout");

      }, ms);
    });

  return Promise.race([
    promise,
    timeoutPromise
  ]);
}

const slowPromise =
  new Promise((resolve) => {

    setTimeout(() => {

      resolve("Finished");

    }, 2000);
  });

withTimeout(slowPromise, 1000)

  .then((res) =>
    console.log(res)
  )

  .catch((err) =>
    console.log(err)
  );
// Promise.race settles with whichever
// promise finishes first.

Try the steps, then compare with ../js/homework.js.

Task 3 Async Loop with for...of
  • Loop through [300, 100, 200].
  • Wait each amount of time sequentially.
  • Confirm execution order matches input order. 
function wait(ms) {

  return new Promise((resolve) => {

    setTimeout(resolve, ms);
  });
}

async function run() {

  for (const ms of [300, 100, 200]) {

    await wait(ms);

    console.log(
      `Finished ${ms}ms`
    );
  }
}

run();
// for...of with await executes
// sequentially in exact order.

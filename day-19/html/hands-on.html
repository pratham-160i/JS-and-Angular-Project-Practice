Task 1 Convert .then() to async/await
  • Convert a Promise chain into async/await.
  • Use try/catch for error handling.
  • Mock APIs using setTimeout(). 
function fetchUser(id) {

  return new Promise((resolve) => {

    setTimeout(() => {

      resolve({
        id,
        name: "Priya"
      });

    }, 300);
  });
}

function fetchOrders(userId) {

  return new Promise((resolve) => {

    setTimeout(() => {

      resolve([
        "Order1",
        "Order2",
        "Order3"
      ]);

    }, 300);
  });
}

async function showOrders(id) {

  try {

    const user =
      await fetchUser(id);

    const orders =
      await fetchOrders(user.id);

    console.log(
      orders.length
    );

  } catch (e) {

    console.error(e);
  }
}

showOrders(7);
// async/await usually reads more
// cleanly than chained .then() calls.

Task 2 Sequential vs Parallel Timing
  • Compare sequential and parallel async execution.
  • Measure execution time. 
function fetchPrice(id) {

  return new Promise((resolve) => {

    setTimeout(() => {

      resolve({
        id,
        price: 100
      });

    }, 500);
  });
}

async function slow() {

  const start = Date.now();

  await fetchPrice(1);

  await fetchPrice(2);

  await fetchPrice(3);

  console.log(
    "Slow:",
    Date.now() - start,
    "ms"
  );
}

async function fast() {

  const start = Date.now();

  await Promise.all([

    fetchPrice(1),

    fetchPrice(2),

    fetchPrice(3)

  ]);

  console.log(
    "Fast:",
    Date.now() - start,
    "ms"
  );
}

slow();

fast();
// Sequential awaits run one after another.
//
// Promise.all runs all promises in parallel.

Task 3 Fix the forEach Trap
  • Observe why forEach(async) behaves unexpectedly.
  • Fix it using for...of and Promise.all(). 
const ids = [1, 2, 3];

function fetchPrice(id) {

  return new Promise((resolve) => {

    setTimeout(() => {

      resolve({
        id,
        price: 100
      });

    }, 500);
  });
}

async function wrong() {

  ids.forEach(async (id) => {

    const result =
      await fetchPrice(id);

    console.log(result);
  });

  console.log(
    "Function ended"
  );
}

wrong();
// Fix 1 — for...of

async function fixedForOf() {

  for (const id of ids) {

    const result =
      await fetchPrice(id);

    console.log(result);
  }
}

fixedForOf();


// Fix 2 — Promise.all

async function fixedPromiseAll() {

  const results =
    await Promise.all(

      ids.map((id) =>
        fetchPrice(id)
      )
    );

  console.log(results);
}

fixedPromiseAll();
// forEach does not wait for async callbacks.
//
// for...of supports await naturally.
//
// Promise.all is best for parallel execution.

Bonus Retry with async/await
  • Build a retry helper for async functions.
  • Retry failed operations multiple times. 
async function retry(fn, attempts) {

  let lastError;

  for (let i = 1; i <= attempts; i++) {

    try {

      return await fn();

    } catch (e) {

      lastError = e;

      console.log(
        `Attempt ${i} failed`
      );
    }
  }

  throw lastError;
}

function flaky() {

  return new Promise((resolve, reject) => {

    const ok =
      Math.random() > 0.5;

    setTimeout(() => {

      if (ok) {

        resolve("Success");

      } else {

        reject(
          new Error("Random failure")
        );
      }

    }, 300);
  });
}

retry(flaky, 3)

  .then((res) =>
    console.log(res)
  )

  .catch((err) =>
    console.log(err.message)
  );
// retry() is useful for unstable APIs,
// network requests, and temporary failures.

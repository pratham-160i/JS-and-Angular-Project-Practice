Day 23 — Iterators & Generators

Lesson

Topic 1 The Iterator Protocol 
// Manually using an array's iterator

const arr = ["a", "b", "c"];

const it =

  arr[Symbol.iterator]();

console.log(it.next());

console.log(it.next());

console.log(it.next());

console.log(it.next());

// for...of is sugar over this protocol

for (const ch of arr) {

  console.log(ch);
}

Arrays, Strings, Maps, Sets, Generators and many built-in objects implement the iterator protocol.

Topic 2 Make an Object Iterable 
const range = {

  from: 1,

  to: 5,

  [Symbol.iterator]() {

    let current = this.from;

    const last = this.to;

    return {

      next() {

        if (current <= last) {

          return {

            value: current++,

            done: false
          };
        }

        return {

          value: undefined,

          done: true
        };
      }
    };
  }
};

for (const n of range)

  console.log(n);

console.log([...range]);

const [first, ...rest] = range;

console.log(first, rest);

Any object becomes iterable when it implements Symbol.iterator.

Topic 3 Generators with function* 
function* simple() {

  yield 1;

  yield 2;

  yield 3;
}

const g = simple();

console.log(g.next());

console.log(g.next());

console.log(g.next());

console.log(g.next());

for (const n of simple())

  console.log(n);

// Range Generator

function* range(from, to) {

  for (

    let i = from;

    i <= to;

    i++
  ) {

    yield i;
  }
}

console.log(

  [...range(1, 5)]
);

// Follow Along

function* g2() {

  yield "a";

  yield "b";
}

const it = g2();

console.log(

  it.next().value
);

console.log(

  it.next().value
);

console.log(

  it.next().done
);

Generators automatically implement the iterator protocol.

Topic 4 Infinite Sequences 
// Infinite ID Generator

function* idGen() {

  let id = 1;

  while (true)

    yield id++;
}

const ids = idGen();

console.log(

  ids.next().value
);

console.log(

  ids.next().value
);

console.log(

  ids.next().value
);

// Infinite Fibonacci

function* fib() {

  let [a, b] = [0, 1];

  while (true) {

    yield a;

    [a, b] =

      [b, a + b];
  }
}

const f = fib();

const first10 = [];

for (

  let i = 0;

  i < 10;

  i++
) {

  first10.push(

    f.next().value
  );
}

console.log(first10);

Infinite generators must always be limited using counters, breaks, or helper functions.

Topic 5 yield* — Delegating 
function* a() {

  yield 1;

  yield 2;
}

function* b() {

  yield 0;

  yield* a();

  yield* [10, 20];

  yield 99;
}

console.log([...b()]);

// Flatten Nested Arrays

function* flatten(items) {

  for (const item of items) {

    if (Array.isArray(item))

      yield* flatten(item);

    else

      yield item;
  }
}

console.log(

  [...flatten(

    [1, [2, [3, [4, 5]]], 6]
  )]
);

yield* delegates iteration to another iterable or generator.

Topic 6 Two-Way Communication: next(value) 
function* dialog() {

  const name =

    yield "What's your name?";

  const age =

    yield `Hi ${name}! How old?`;

  return `${name}, ${age}, recorded.`;
}

const g = dialog();

console.log(

  g.next().value
);

console.log(

  g.next("Priya").value
);

console.log(

  g.next(25).value
);

Values passed to next(value) become the result of the previous yield expression.

Topic 7 Pagination Generator 
function fetchPage(page) {

  const data = {

    1: ["pen", "book", "bag"],

    2: ["mug", "lamp", "fan"],

    3: ["chair", "desk", "rug"],

    4: ["plant", "vase"]
  };

  return Promise.resolve(

    data[page] || []
  );
}

async function* paginate() {

  let page = 1;

  while (true) {

    const items =

      await fetchPage(page);

    if (

      items.length === 0
    ) {

      return;
    }

    yield* items;

    page++;
  }
}

(async () => {

  for await (

    const item of paginate()
  ) {

    console.log(item);
  }

})();

Async generators combine async/await with iterators and are consumed using for-await-of.

Task 1 Deep Immutable Update
  • Create a deeply nested object.
  • Perform a single immutable update on a leaf value.
  • Verify unchanged branches still share references. 
const state = {

  user: {

    profile: {

      address: {

        city: "Jaipur",

        zip: 302001
      }
    },

    settings: {

      theme: "light"
    }
  }
};

const updatedState = {

  ...state,

  user: {

    ...state.user,

    profile: {

      ...state.user.profile,

      address: {

        ...state.user.profile.address,

        city: "Mumbai"
      }
    }
  }
};

console.log(
  updatedState.user.profile.address.city
);

console.log(

  state.user.settings ===
  updatedState.user.settings
);

console.log(

  state.user.profile ===
  updatedState.user.profile
);
// Unchanged branches still
// share references.
//
// Changed branches receive
// new objects.

Task 2 Implement deepFreeze()
  • Build deepFreeze(obj) recursively.
  • Freeze nested objects completely.
  • Test with a 3-level object. 
function deepFreeze(obj) {

  Object.keys(obj).forEach((key) => {

    if (

      typeof obj[key] === "object"

      && obj[key] !== null

    ) {

      deepFreeze(obj[key]);
    }
  });

  return Object.freeze(obj);
}

const user = {

  name: "Priya",

  profile: {

    city: "Jaipur",

    prefs: {

      theme: "dark"
    }
  }
};

deepFreeze(user);

user.profile.city = "Delhi";

console.log(user);
// Object.freeze only freezes
// one level by default.
//
// deepFreeze recursively freezes
// all nested objects.

Task 4 Build pick(obj, keys)
  • Create a helper that extracts selected keys.
  • Return a new object only with requested properties.
  • Use destructuring and computed keys. 
function pick(obj, keys) {

  const result = {};

  keys.forEach((key) => {

    if (key in obj) {

      result[key] = obj[key];
    }
  });

  return result;
}

const user = {

  id: 1,

  name: "Priya",

  city: "Jaipur",

  role: "Developer"
};

const picked =

  pick(user, [

    "name",

    "role"
  ]);

console.log(picked);
// pick() creates a smaller,
// controlled object copy.
//
// Useful for APIs and state shaping.

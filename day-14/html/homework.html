Problem 7 Part 7 
function createAccount(intial) {
    let bal = intial;
    let transCount = 0;
    return {
        deposit(amt) {
            bal += amt
            transCount++;
        },
        withdraw(amt) {
            bal -= amt
            transCount++;
        },
        getBalance() {
            return bal
        },
        getCount() {
            return transCount
        }
    }
}

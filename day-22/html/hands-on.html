Day 22 — Map / Set / WeakMap / WeakSet

Hands-on

Task 1 Map vs Object
  • Create a Map of products and prices.
  • Use .has(), .get(), .delete(), and .size.
  • Convert between Map and Object. 
const products = new Map([

  ["pen", 50],

  ["book", 200],

  ["bag", 800]
]);

for (const [item, price] of products) {

  console.log(
    `${item}: ₹${price}`
  );
}

console.log(
  products.has("pen")
);

console.log(
  products.get("book")
);

products.delete("pen");

console.log(
  products.size
);


const obj =

  Object.fromEntries(products);

console.log(obj);


const backToMap =

  new Map(

    Object.entries(obj)
  );

console.log(backToMap);
// Map preserves insertion order
// and supports any key type.
//
// Object.fromEntries converts
// Map → Object.
//
// Object.entries converts
// Object → Map.

Task 2 Deduplicate with Set
  • Remove duplicates using Set.
  • Convert back into an array.
  • Test mixed value types. 
const ids = [

  101,

  102,

  103,

  101,

  104,

  102,

  105
];

const uniqueIds =

  [...new Set(ids)];

console.log(uniqueIds);

console.log(
  uniqueIds.length
);


const mixed = [

  1,

  "1",

  1,

  true,

  1n
];

const mixedSet =

  [...new Set(mixed)];

console.log(mixedSet);
// Set stores only unique values.
//
// Number 1 and string "1"
// are different values.
//
// BigInt 1n is also unique.

Task 3 Cache with Map
  • Build memoize(fn) using a Map.
  • Cache expensive calculations.
  • Verify cached results reuse old values. 
function memoize(fn) {

  const cache = new Map();

  return function (n) {

    if (cache.has(n)) {

      return cache.get(n);
    }

    const result = fn(n);

    cache.set(n, result);

    console.log(
      "Cache Size:",
      cache.size
    );

    return result;
  };
}

function expensiveSquare(n) {

  console.log("computing...");

  return n * n;
}

const memoSquare =

  memoize(expensiveSquare);

console.log(
  memoSquare(5)
);

console.log(
  memoSquare(5)
);

console.log(
  memoSquare(10)
);
// Map is ideal for caching.
//
// Cache prevents repeated
// expensive computations.

Bonus WeakMap for Private Data
  • Store metadata externally using WeakMap.
  • Avoid modifying original objects.
  • Understand garbage collection behavior. 
const metadata =

  new WeakMap();

function attach(obj, data) {

  metadata.set(obj, data);
}

function get(obj) {

  return metadata.get(obj);
}

let button1 = {

  id: "btn1"
};

let button2 = {

  id: "btn2"
};

attach(button1, {

  lastClick: Date.now()
});

attach(button2, {

  lastClick: Date.now()
});

console.log(
  get(button1)
);

console.log(
  get(button2)
);


// Remove one reference

button1 = null;
// WeakMap keys must be objects.
//
// When an object reference
// disappears, its WeakMap entry
// becomes eligible for
// garbage collection.

Task 1 Build a Tag Counter with Map
  • Build a tag frequency counter using Map.
  • Each post contains a .tags array.
  • Return a tag → count mapping. 
const posts = [

  {

    id: 1,

    tags: ["js", "react"]
  },

  {

    id: 2,

    tags: ["js", "node"]
  },

  {

    id: 3,

    tags: ["react", "css"]
  }
];

function tagCounter(posts) {

  const map = new Map();

  for (const post of posts) {

    for (const tag of post.tags) {

      map.set(

        tag,

        (map.get(tag) || 0) + 1
      );
    }
  }

  return map;
}

const result =
  tagCounter(posts);

console.log(result);
// Map is ideal for counters
// because keys can be managed
// efficiently with insertion order.

Task 2 Build union(), intersection(), difference()
  • Create reusable set helper functions.
  • Use Set and spread syntax.
  • Return new sets without mutation. 
function union(a, b) {

  return new Set([

    ...a,

    ...b
  ]);
}

function intersection(a, b) {

  return new Set(

    [...a].filter(

      (value) => b.has(value)
    )
  );
}

function difference(a, b) {

  return new Set(

    [...a].filter(

      (value) => !b.has(value)
    )
  );
}

const setA =

  new Set([1, 2, 3]);

const setB =

  new Set([3, 4, 5]);

console.log(

  union(setA, setB)
);

console.log(

  intersection(setA, setB)
);

console.log(

  difference(setA, setB)
);
// union:
// combines all unique values.
//
// intersection:
// keeps common values.
//
// difference:
// removes overlapping values.

Try the steps, then compare with ../js/homework.js.

Task 3 Sort a Map by Age
  • Convert a Map into an array.
  • Sort entries by age.
  • Use destructuring. 
const ages = new Map([

  ["Priya", 24],

  ["Aarav", 21],

  ["Riya", 27]
]);

const sorted =

  [...ages.entries()]

    .sort(([, ageA], [, ageB]) => {

      return ageA - ageB;
    });

console.log(sorted);
// entries() converts Map
// into [key, value] pairs.
//
// Destructuring extracts
// ages directly during sorting.

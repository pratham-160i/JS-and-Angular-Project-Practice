Topic 1 Debounce vs Throttle

Debounce waits until activity stops. Throttle limits execution frequency. :contentReference[oaicite:0]{index=0} 

// DEBOUNCE

function debounce(

  fn,

  delay
) {

  let timer;

  return function (...args) {

    clearTimeout(timer);

    timer = setTimeout(

      () =>

        fn.apply(

          this,

          args
        ),

      delay
    );
  };
}

// THROTTLE

function throttle(

  fn,

  delay
) {

  let lastCall = 0;

  return function (...args) {

    const now =

      Date.now();

    if (

      now - lastCall < delay
    ) {

      return;
    }

    lastCall = now;

    fn.apply(

      this,

      args
    );
  };
}

// Debounced Search

const handleSearch =

  debounce((q) => {

    console.log(

      "searching for:",

      q
    );

  }, 300);

// Throttled Scroll

const onScroll =

  throttle(() => {

    console.log(

      "scroll at",

      window.scrollY
    );

  }, 100);
// FOLLOW ALONG

const log = debounce(

  () =>

    console.log("fired"),

  100
);

log();

log();

log();

// wait 200ms

log();

// Output:
//
// fired
// fired
//
// Total Fires = 2

Debounce: search boxes. Throttle: scrolling and dragging. :contentReference[oaicite:1]{index=1}

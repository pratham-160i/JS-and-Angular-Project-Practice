Task 3 Use Debounce and Promise.all

Use debounce for user input and Promise.all for parallel API requests. 

// debounce.js

export function debounce(

    fn,

    delay
) {

    let timer;

    return (...args) => {

        clearTimeout(timer);

        timer = setTimeout(

            () => fn(...args),

            delay
        );
    };
}

async function loadData() {

    const [

        user,

        repos

    ] = await Promise.all([

        fetch(

            "https://api.github.com/users/octocat"
        ).then(

            r => r.json()
        ),

        fetch(

            "https://api.github.com/users/octocat/repos"
        ).then(

            r => r.json()
        )
    ]);

    console.log(user);

    console.log(repos);
}

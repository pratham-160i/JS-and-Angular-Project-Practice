Task 1 Implement debounce

Write debounce(fn, delay) from scratch and use it on a fake search. 

function debounce(

    fn,

    delay
) {

    let timer;

    return function (...args) {

        clearTimeout(timer);

        timer = setTimeout(

            () =>

                fn(...args),

            delay
        );
    };
}

function handleSearch(

    query
) {

    console.log(

        "Searching:",

        query
    );
}

const debouncedSearch =

    debounce(

        handleSearch,

        300
    );

// First burst

for (

    let i = 1;

    i <= 5;

    i++
) {

    setTimeout(

        () =>

            debouncedSearch(

                `query${i}`
            ),

        i * 50
    );
}

// Second burst

setTimeout(() => {

    debouncedSearch(

        "query6"
    );

    setTimeout(

        () =>

            debouncedSearch(

                "query7"
            ),

        50
    );

}, 750);

Expected Output:
Searching: query5
Searching: query7

Task 2 Implement throttle

Write throttle(fn, delay) from scratch and simulate scrolling. 

function throttle(

    fn,

    delay
) {

    let lastCall = 0;

    return function (...args) {

        const now =

            Date.now();

        if (

            now - lastCall < delay
        ) {

            return;
        }

        lastCall = now;

        fn(...args);
    };
}

const throttledScroll =

    throttle(

        () => {

            console.log(

                "Scroll event"
            );

        },

        1000
    );

for (

    let i = 0;

    i < 10;

    i++
) {

    throttledScroll();
}

Expected Output:
Scroll event

(Only one call because all 10 executions happen instantly)

Task 3 Spot the Memory Leaks

Identify whether each snippet is safe or causes a memory leak. 

// (a)

setInterval(

    updateClock,

    1000
);

// Leak

// Interval continues running
// after component unmounts.



// (b)

const handler = () =>

    doStuff();

btn.addEventListener(

    "click",

    handler
);

btn.remove();

// Leak

// Event listener still
// references the DOM node.



// (c)

function test() {

    msg = "hello";
}

// Leak

// Creates accidental
// global variable.



// (d)

useEffect(() => {

    const id =

        setInterval(

            updateClock,

            1000
        );

    return () =>

        clearInterval(id);

}, []);

// Safe

// Cleanup removes
// interval correctly.

Results:
(a) Leak
(b) Leak
(c) Leak
(d) Safe

Bonus Sanitise User Input

Strip HTML tags and safely display user input. 

function safeText(

    html
) {

    const div =

        document.createElement(

            "div"
        );

    div.innerHTML = html;

    return div.textContent ||

           div.innerText ||

           "";
}

const input =

    '<img src=x onerror="alert(1)"> hello';

const cleanText =

    safeText(input);

console.log(

    cleanText
);

const output =

    document.getElementById(

        "output"
    );

output.textContent =

    cleanText;

Expected Output:
hello

No JavaScript executes because textContent renders plain text.

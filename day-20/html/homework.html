Task 2 Build chunk(array, fn, chunkSize)
  • Process a large array in chunks.
  • Yield control back to the event loop between chunks.
  • Use setTimeout(0). 
function chunk(array, fn, chunkSize) {

  let index = 0;

  function processChunk() {

    const end =
      Math.min(
        index + chunkSize,
        array.length
      );

    while (index < end) {

      fn(array[index]);

      index++;
    }

    if (index < array.length) {

      setTimeout(
        processChunk,
        0
      );
    }
  }

  processChunk();
}

const bigArray =
  Array.from(
    { length: 100000 },
    (_, i) => i
  );

chunk(

  bigArray,

  (item) => {

    if (item % 25000 === 0) {

      console.log(
        "Processed:",
        item
      );
    }
  },

  5000
);
// Chunking prevents the UI
// and event loop from freezing
// during heavy processing.

Task 3 Prove await Uses Microtasks
  • Schedule both await and setTimeout(0).
  • Predict which runs first.
  • Verify the event loop behavior. 
async function test() {

  console.log("Start");

  await Promise.resolve();

  console.log("After await");
}

test();

setTimeout(() => {

  console.log("Timer");

}, 0);

console.log("End");
// Output:
//
// Start
// End
// After await
// Timer
//
// await resumes execution
// as a microtask.
//
// Microtasks always run before
// macrotasks like setTimeout.

Task 1 Implement compose from Scratch
  • Implement compose(...fns) that runs functions right-to-left.
  • Use addOne, square, and negate.
  • Compare the output ordering with pipe(). 
function compose(...fns) {

    return function(value) {

        return fns.reduceRight(

            (acc, fn) =>

                fn(acc),

            value
        );
    };
}

const addOne =

    (x) => x + 1;

const square =

    (x) => x * x;

const negate =

    (x) => -x;

const result =

    compose(

        negate,

        square,

        addOne
    );

console.log(

    result(5)
);

Output:
36

compose() executes:
addOne → square → negate

Task 2 Refactor Mutating Function to Pure Function
  • Convert a mutating array function into a pure version.
  • Verify the original array remains unchanged. 
// Impure Version

function addItem(

    arr,

    item
) {

    arr.push(item);

    return arr;
}

const nums =

    [1, 2, 3];

addItem(nums, 4);

console.log(nums);

// Pure Version

function addItemPure(

    arr,

    item
) {

    return [

        ...arr,

        item
    ];
}

const original =

    [1, 2, 3];

const updated =

    addItemPure(

        original,

        4
    );

console.log(original);

console.log(updated);

Output:
Original: [1, 2, 3]
Updated: [1, 2, 3, 4]

Task 3 Implement curry(fn)
  • Create a generic curry() utility.
  • Test with curry(add)(1)(2)(3). 
function curry(fn) {

    return function curried(

        ...args
    ) {

        if (

            args.length >=

            fn.length
        ) {

            return fn(

                ...args
            );
        }

        return function(

            ...nextArgs
        ) {

            return curried(

                ...args,

                ...nextArgs
            );
        };
    };
}

function add(

    a,

    b,

    c
) {

    return a + b + c;
}

const curriedAdd =

    curry(add);

console.log(

    curriedAdd(1)(2)(3)
);

Output:
6

Task 4 Average Order Value Pipeline
  • Use the order data from Topic 7.
  • Build a pipeline that returns the average order value.
  • Exclude GST.
  • Use only small pure functions. 
const orders = [

    {

        id: 1,

        item: "Pen",

        price: 50,

        quantity: 2
    },

    {

        id: 2,

        item: "Book",

        price: 200,

        quantity: 1
    },

    {

        id: 3,

        item: "Bag",

        price: 800,

        quantity: 1
    },

    {

        id: 4,

        item: "Mug",

        price: 150,

        quantity: 3
    }
];

function pipe(...fns) {

    return function(value) {

        return fns.reduce(

            (acc, fn) =>

                fn(acc),

            value
        );
    };
}

const lineTotals =

    (orders) =>

        orders.map(

            (o) =>

                o.price *

                o.quantity
        );

const sum =

    (values) =>

        values.reduce(

            (a, b) =>

                a + b,

            0
        );

const average =

    (total) =>

        total /

        orders.length;

const averageOrderValue =

    pipe(

        lineTotals,

        sum,

        average
    );

console.log(

    averageOrderValue(
        orders
    )
);

Output:
387.5

Average order value before GST.

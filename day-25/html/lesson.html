Topic 1 Pure Functions

A function is pure when it always produces the same output for the same input and has no side effects. :contentReference[oaicite:0]{index=0} 

// PURE

function add(a, b) {

  return a + b;
}

function double(x) {

  return x * 2;
}

function withGST(

  price,

  rate = 18
) {

  return price *

    (1 + rate / 100);
}

// IMPURE

let multiplier = 2;

function impureDouble(x) {

  return x * multiplier;
}

// IMPURE

function logAndDouble(x) {

  console.log(x);

  return x * 2;
}

// IMPURE

function addItem(

  arr,

  item
) {

  arr.push(item);

  return arr;
}

// PURE VERSION

function addItemPure(

  arr,

  item
) {

  return [

    ...arr,

    item
  ];
}

Same input → same output. No side effects. :contentReference[oaicite:1]{index=1}

// once()

function once(fn) {

  let called = false;

  let result;

  return (...args) => {

    if (called)

      return result;

    called = true;

    return (

      result = fn(...args)
    );
  };
}

const init =

  once(() =>

    "initialised"
  );

console.log(init());

console.log(init());

Topic 4 compose() and pipe() 
const compose =

  (...fns) =>

    (x) =>

      fns.reduceRight(

        (acc, fn) =>

          fn(acc),

        x
      );

const pipe =

  (...fns) =>

    (x) =>

      fns.reduce(

        (acc, fn) =>

          fn(acc),

        x
      );

const trim =

  (s) => s.trim();

const lower =

  (s) =>

    s.toLowerCase();

const split =

  (s) =>

    s.split(/\s+/);

const wordCount =

  (arr) =>

    arr.length;

const countWords =

  pipe(

    trim,

    lower,

    split,

    wordCount
  );

  countWords(

Topic 5 Currying

  a,

  b,

  c
) {

  return a + b + c;
}

function addCurried(a) {

  return function (b) {

    return function (c) {

      return a + b + c;
    };
  };
}

addCurried(

Topic 7 Real Pipeline Example 
const orders = [

  {
    id: 1,
    item: "Pen",
    price: 50,
    quantity: 2
  },

  {
    id: 2,
    item: "Book",
    price: 200,
    quantity: 1
  },

  {
    id: 3,
    item: "Bag",
    price: 800,
    quantity: 1
  },

  {
    id: 4,
    item: "Mug",
    price: 150,
    quantity: 3
  }
];

const lineTotal =

  (o) =>

    o.price *

    o.quantity;

const withGST =

  (amt) =>

    amt * 1.18;

const round2 =

  (n) =>

    Math.round(
      n * 100
    ) / 100;

const formatINR =

  (n) =>

    `₹${n.toLocaleString(
      "en-IN"
    )}`;

const sum =

  (a, b) => a + b;

const grandTotal =

  pipe(

    (orders) =>

      orders.map(
        lineTotal
      ),

    (totals) =>

      totals.reduce(
        sum,
        0
      ),

    withGST,

    round2,

    formatINR
  );

console.log(

  grandTotal(
    orders
  )
);

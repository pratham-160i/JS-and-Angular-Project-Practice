Topic 1 Named vs Default Exports

JavaScript modules support both named exports and default exports. 

// utils.js — named exports

export function formatPrice(p) {

  return `₹${p}`;
}

export function gst(

  p,

  rate = 18
) {

  return p * rate / 100;
}

export const TAX_RATE = 18;

// main.js

import {

  formatPrice,

  gst,

  TAX_RATE

} from "./utils.js";

// Card.js

export default function Card({

  title

}) {

  /* ... */
}

// main.js

import Card

  from "./Card.js";

import Whatever

  from "./Card.js";

Named exports are preferred for utilities. Default exports are useful when a file has one primary feature.

Topic 2 Aliasing & Namespace Imports 
// Rename on import

import {

  formatPrice as fp

} from "./utils.js";

fp(100);

// Rename on export

export {

  formatPrice as format

} from "./utils.js";

// Namespace import

import * as utils

  from "./utils.js";

console.log(

  utils.formatPrice(100)
);

console.log(

  utils.gst(100)
);

console.log(

  utils.TAX_RATE
);

// Follow Along

// utils.js

export const a = 1;

export const b = 2;

// main.js

import * as u

  from "./utils.js";

console.log(

  u.a + u.b
);

console.log(

  u.c
);

First output: 3
Second output: undefined

Topic 3 Barrels — Single Import Path 
// components/index.js

export {

  default as Button

} from "./Button.js";

export {

  default as Card

} from "./Card.js";

export {

  default as Modal

} from "./Modal.js";

// Re-export named exports

export * from "./hooks.js";

export {

  useForm,

  useDebounce

} from "./hooks.js";

// App.js

import {

  Button,

  Card,

  Modal

} from "./components";

Barrels simplify imports by providing a single entry point.

Topic 4 Dynamic import() 
// Static Import

import {

  gst

} from "./utils.js";

// Dynamic Import

async function showSettings() {

  const {

    Settings

  } = await import(

    "./Settings.js"
  );

  Settings.render();
}

// Conditional Load

button.addEventListener(

  "click",

  async () => {

    const Chart =

      await import(

        "./BigChartLibrary.js"
      );

    Chart.draw();
  }
);

// React Lazy Loading

const Settings =

  React.lazy(() =>

    import("./Settings.js")
  );

Dynamic imports enable lazy loading and code splitting.

Topic 5 Live Bindings 
// counter.js

export let count = 0;

export function inc() {

  count++;
}

// main.js

import {

  count,

  inc

} from "./counter.js";

console.log(count);

inc();

console.log(count);

// Singleton Pattern

// cache.js

const cache =

  new Map();

export default cache;

// Anywhere

import cache

  from "./cache.js";

cache.set(

  "key",

  "value"
);

Imported bindings stay connected to the original exported value and update automatically.

Topic 6 Circular Dependencies 
// a.js

import {

  fromB

} from "./b.js";

export const fromA =

  "I am A";

console.log(

  "a sees fromB =",

  fromB
);

// b.js

import {

  fromA

} from "./a.js";

export const fromB =

  "I am B";

console.log(

  "b sees fromA =",

  fromA
);

// Possible Fixes

// 1. Extract shared logic
//    into a third module

// 2. Read values lazily
//    inside functions

// 3. Use ESLint or
//    TypeScript checks

Circular dependencies often cause undefined values during module initialization.

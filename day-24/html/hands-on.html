Day 24 — ES Modules in Depth

Hands-on

Task 1 Multiple Named Exports + Barrel

Create three modules that export named functions and use a barrel file to re-export them from a single location.

  • Create add.js, multiply.js and divide.js.
  • Export a named function from each file.
  • Create math/index.js to re-export all functions.
  • Import all three functions from a single line in app.js.
// math/add.js

export function add(a, b) {

    return a + b;
}

// math/multiply.js

export function multiply(a, b) {

    return a * b;
}

// math/divide.js

export function divide(a, b) {

    return a / b;
}

// math/index.js

export { add }

    from "./add.js";

export { multiply }

    from "./multiply.js";

export { divide }

    from "./divide.js";

// app.js

import {

    add,

    multiply,

    divide

} from "./math/index.js";

console.log(

    add(10, 5)
);

console.log(

    multiply(10, 5)
);

console.log(

    divide(10, 5)
);

Output:
15
50
2

Task 2 Default + Named in Same File

Export both a default value and a named value from the same module. 

// Button.js

export const ButtonStyles = {

    primary: "btn-primary",

    secondary: "btn-secondary"
};

export default function Button(

    title
) {

    console.log(

        `Button:
${title}`
    );
}

// app.js

import Button, {

    ButtonStyles

} from "./Button.js";

Button("Save");

console.log(

    ButtonStyles.primary
);

Output:
Button: Save
btn-primary

Task 3 Dynamic import()

Dynamically load a module only when needed. 

// heavy.js

export function compute() {

    return "computed!";
}

// main.js

async function loadAndRun() {

    const module =

        await import(

            "./heavy.js"
        );

    console.log(

        module.compute()
    );
}

loadAndRun();

Output:
computed!

In a real Vite/Webpack build, heavy.js is downloaded only when loadAndRun() executes.

Bonus Singleton Cache

Share one cache instance across multiple modules. 

// cache.js

const cache =

    new Map();

export default cache;

// userService.js

import cache

    from "./cache.js";

cache.set(

    1,

    {

        id: 1,

        name: "Priya"
    }
);

// productService.js

import cache

    from "./cache.js";

cache.set(

    "P101",

    {

        id: "P101",

        name: "Book"
    }
);

// main.js

import "./userService.js";

import "./productService.js";

import cache

    from "./cache.js";

console.log(

    cache
);

Output:
Map(2) { 1 => { id: 1, name: "Priya" }, "P101" => { id: "P101", name: "Book" } }

Both services share the same cache instance.

Task 1 Barrel File with Named Re-exports and export *
  • Create a folder containing three modules.
  • Build a barrel index.js using named re-exports and export *. 
// math/add.js

export function add(a, b) {

    return a + b;
}

// math/subtract.js

export function subtract(a, b) {

    return a - b;
}

// math/constants.js

export const PI = 3.14159;

export const TAX = 18;

// math/index.js

export {

    add

} from "./add.js";

export {

    subtract

} from "./subtract.js";

export * from "./constants.js";

// app.js

import {

    add,

    subtract,

    PI,

    TAX

} from "./math/index.js";

console.log(

    add(10, 5)
);

console.log(

    subtract(10, 5)
);

console.log(PI);

console.log(TAX);

Output:
15
5
3.14159
18

Task 2 Dynamic import() in Vite
  • Add a dynamic import to load a feature only when required.
  • Verify the chunk loads in DevTools Network tab. 
// feature.js

export function showFeature() {

    console.log(

        "Feature Loaded!"
    );
}

// main.js

async function loadFeature() {

    const module =

        await import(

            "./feature.js"
        );

    module.showFeature();
}

document

    .getElementById(

        "loadBtn"
    )

    .addEventListener(

        "click",

        loadFeature
    );

Expected Result:
feature.js is downloaded only after the button click.

Task 3 Circular Dependency
  • Create a circular dependency intentionally.
  • Observe the bug.
  • Refactor to remove the cycle. 
// a.js

import {

    valueB

} from "./b.js";

export const valueA =

    "A";

console.log(

    "A sees:",

    valueB
);

// b.js

import {

    valueA

} from "./a.js";

export const valueB =

    "B";

console.log(

    "B sees:",

    valueA
);

One module may see undefined because initialization order becomes circular. 

// Refactored Version

// shared.js

export const sharedValue =

    "Shared";

// a.js

import {

    sharedValue

} from "./shared.js";

console.log(

    "A:",

    sharedValue
);

// b.js

import {

    sharedValue

} from "./shared.js";

console.log(

    "B:",

    sharedValue
);

Fixed:
Both modules depend on shared.js instead of depending on each other.

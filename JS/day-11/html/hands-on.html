Task 1: Click Counter

What to do:

  • Grab the #my-btn button and the #click-counter paragraph
  • On every click, increment a counter variable and update the paragraph's textContent
  • Bonus: after 10 clicks, change the button's text to "Stop clicking!"

Total Clicks: 0

let myBtn = document.querySelector("#my-btn");
let clickCounter = document.querySelector("#click-counter");
let taskClicks = 0;

myBtn.addEventListener("click", () => {
  taskClicks++;
  clickCounter.textContent = "Total Clicks: " + taskClicks;
  console.log("Button clicked!");
  
  // Bonus: after 10 clicks
  if (taskClicks === 10) {
    myBtn.textContent = "Stop clicking!";
  }
});

// Multiple listeners on same button
myBtn.addEventListener("click", () => {
  console.log("Second handler also fires");
});

Task 2: Live Input Preview

What to do:

  • Grab the #name-input input field and #name-display paragraph
  • On every keystroke (use the input event), update the preview with the current value
  • Bonus: if the input is empty, show "Start typing..." in the preview

You typed: 

let nameInput = document.querySelector("#name-input");
let nameDisplay = document.querySelector("#name-display");

nameInput.addEventListener("input", (e) => {
  if (e.target.value === "") {
    nameDisplay.textContent = "Start typing...";
  } else {
    nameDisplay.textContent = "You typed: " + e.target.value;
  }
  console.log("User typed:", e.target.value);
});

Task 3: Event Delegation on List

What to do:

  • Grab the #todo-list parent element
  • Attach ONE click listener that toggles the done class on whichever <li> was clicked
  • Use e.target.tagName === "LI" to filter clicks to only list items
  • Bonus: programmatically add a new <li> and verify clicking it also toggles correctly
  • Buy milk (click to mark)
  • Walk dog (click to mark)
  • Pay bills (click to mark)
let todoList = document.querySelector("#todo-list");

// Single listener on parent (Event Delegation)
todoList.addEventListener("click", (e) => {
  if (e.target.tagName === "LI") {
    let id = e.target.getAttribute("data-id");
    console.log("Clicked item with id:", id);
    e.target.classList.toggle("done");
    
    // Visual feedback
    if (e.target.style.backgroundColor === "rgb(144, 238, 144)") {
      e.target.style.backgroundColor = "white";
      e.target.style.textDecoration = "none";
    } else {
      e.target.style.backgroundColor = "#90EE90";
      e.target.style.textDecoration = "line-through";
    }
  }
});

// Bonus: add new item programmatically
let newItem = document.createElement("li");
newItem.textContent = "New task (added dynamically)";
newItem.setAttribute("data-id", "4");
newItem.style.padding = "8px";
newItem.style.margin = "5px 0";
newItem.style.background = "white";
newItem.style.borderRadius = "4px";
newItem.style.cursor = "pointer";
newItem.style.borderLeft = "4px solid #2196F3";
todoList.appendChild(newItem);

// Clicking the new item also toggles - no extra wiring needed!

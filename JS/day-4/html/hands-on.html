Bonus: Guard Clauses Refactor

What to do:

  • Refactor the nested function below using guard clauses — flip each condition and return early
  • Test with: a valid user, a user with isBanned = true, and null
// BEFORE: Nested if statements (hard to read)
function canPostComment_old(user) {
  if (user) {
    if (!user.isBanned) {
      if (user.email) {
        console.log("Can post comment");
        return true;
      }
    }
  }
}

// AFTER: Guard clauses (cleaner, early return)
function canPostComment(user) {
  // Guard: user must exist
  if (!user) {
    console.log("User does not exist");
    return false;
  }
  
  // Guard: user must not be banned
  if (user.isBanned) {
    console.log("User is banned");
    return false;
  }
  
  // Guard: user must have email
  if (!user.email) {
    console.log("User has no email");
    return false;
  }
  
  // If all guards pass
  console.log("Can post comment");
  return true;
}

// Test cases
const validUser = { isBanned: false, email: "user@example.com" };
canPostComment(validUser);              // Can post comment

const bannedUser = { isBanned: true, email: "bad@example.com" };
canPostComment(bannedUser);             // User is banned

canPostComment(null);                   // User does not exist

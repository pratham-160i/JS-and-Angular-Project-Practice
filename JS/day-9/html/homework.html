Exercise 1: Rest Parameters with joinNames

What to do:

  • Write a function joinNames(separator, ...names) that joins all names using the separator
  • Test: joinNames(", ", "Priya", "Aarav", "Riya") should give "Priya, Aarav, Riya"
function joinNames(separator, ...names) {
  return names.join(separator);
}

console.log(joinNames(", ", "Priya", "Aarav", "Riya"));
// "Priya, Aarav, Riya"

console.log(joinNames(" - ", "Alice", "Bob", "Charlie"));
// "Alice - Bob - Charlie"

Exercise 2: Merge Settings with Spread

What to do:

  • Declare user defaults: { theme: "light", fontSize: 14, lang: "en" }
  • Declare user settings: { theme: "dark", fontSize: 18 }
  • Use object spread to merge them so user settings override defaults
  • Log the final merged object
let userDefaults = { theme: "light", fontSize: 14, lang: "en" };
let userSettings = { theme: "dark", fontSize: 18 };
let final = { ...userDefaults, ...userSettings };

console.log("Final settings: ", final);
// { theme: "dark", fontSize: 18, lang: "en" }

Exercise 3: Nested Destructuring

What to do:

  • Use nested destructuring to extract name and nested math grade from a student object
  • Log the result: " scored in math"
let student = {
  id: 101,
  name: "Pratham",
  grades: {
    math: 95,
    english: 88
  }
};

let { name, grades: { math } } = student;
console.log(`${name} scored ${math} in math`);
// Pratham scored 95 in math

Task 1 EventEmitter with once()

Extend createEmitter() by adding a once(event, fn) method that automatically unsubscribes after the first emit. 

function createEmitter() {

    const listeners =

        new Map();

    return {

        on(event, fn) {

            if (

                !listeners.has(event)
            ) {

                listeners.set(

                    event,

                    new Set()
                );
            }

            listeners

                .get(event)

                .add(fn);
        },

        off(event, fn) {

            listeners

                .get(event)

                ?.delete(fn);
        },

        once(event, fn) {

            const wrapper =

                (...args) => {

                    fn(...args);

                    this.off(

                        event,

                        wrapper
                    );
                };

            this.on(

                event,

                wrapper
            );
        },

        emit(event, ...args) {

            listeners

                .get(event)

                ?.forEach(

                    (fn) =>

                        fn(...args)
                );
        }
    };
}

const emitter =

    createEmitter();

emitter.once(

    "hello",

    () =>

        console.log(

            "Runs Once"
        )
);

emitter.emit("hello");

emitter.emit("hello");

Output:
Runs Once

Task 2 Cache Factory with TTL

Create a cache where each entry expires automatically after its configured TTL. 

function cacheFactory(ttl) {

    const cache =

        new Map();

    return {

        set(key, value) {

            cache.set(

                key,

                {

                    value,

                    expires:

                        Date.now()

                        + ttl
                }
            );
        },

        get(key) {

            const item =

                cache.get(key);

            if (

                !item
            ) {

                return undefined;
            }

            if (

                Date.now() >

                item.expires
            ) {

                cache.delete(key);

                return undefined;
            }

            return item.value;
        },

        has(key) {

            return (

                this.get(key)

                !== undefined
            );
        }
    };
}

const cache =

    cacheFactory(

        3000
    );

cache.set(

    "user",

    "Priya"
);

console.log(

    cache.get("user")
);

setTimeout(() => {

    console.log(

        cache.get("user")
    );

}, 4000);

Output:
Priya
undefined

(after 4 seconds)

Task 3 Debug Array with Proxy

Wrap an array in a Proxy and log every read, write and delete operation. 

const arr =

    [10, 20, 30];

const debugArray =

    new Proxy(

        arr,

        {

            get(

                target,

                prop
            ) {

                console.log(

                    "READ:",

                    prop
                );

                return target[prop];
            },

            set(

                target,

                prop,

                value
            ) {

                console.log(

                    "WRITE:",

                    prop,

                    "=",

                    value
                );

                target[prop] =

                    value;

                return true;
            },

            deleteProperty(

                target,

                prop
            ) {

                console.log(

                    "DELETE:",

                    prop
                );

                delete target[prop];

                return true;
            }
        }
    );

console.log(

    debugArray[0]
);

debugArray[1] = 99;

delete debugArray[2];

Output:
READ: 0
10
WRITE: 1 = 99
DELETE: 2

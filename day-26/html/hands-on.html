Task 1 Tiny EventEmitter

Build createEmitter() returning on, off and emit methods. 

function createEmitter() {

    const listeners =

        new Map();

    return {

        on(event, fn) {

            if (

                !listeners.has(event)
            ) {

                listeners.set(

                    event,

                    new Set()
                );
            }

            listeners

                .get(event)

                .add(fn);
        },

        off(event, fn) {

            listeners

                .get(event)

                ?.delete(fn);
        },

        emit(event, ...args) {

            listeners

                .get(event)

                ?.forEach(

                    (fn) =>

                        fn(...args)
                );
        }
    };
}

const emitter =

    createEmitter();

function listener1() {

    console.log(

        "Listener 1"
    );
}

function listener2() {

    console.log(

        "Listener 2"
    );
}

emitter.on(

    "hello",

    listener1
);

emitter.on(

    "hello",

    listener2
);

emitter.emit("hello");

emitter.off(

    "hello",

    listener1
);

emitter.emit("hello");

Output:
Listener 1
Listener 2

Listener 2

Task 2 User Factory

Create a user factory with a default role. 

function createUser(

    name,

    role = "user"
) {

    return {

        name,

        role,

        canEdit() {

            return (

                role ===
                "admin"
            );
        }
    };
}

const user1 =

    createUser(
        "Priya"
    );

const user2 =

    createUser(

        "Aarav",

        "admin"
    );

console.log(

    user1.canEdit()
);

console.log(

    user2.canEdit()
);

Output:
false
true

Task 3 Module-Singleton Cache

Create independent caches and then demonstrate a shared singleton cache. 

function createCache() {

    const cache =

        new Map();

    return {

        get(key) {

            return cache.get(
                key
            );
        },

        set(

            key,

            value
        ) {

            cache.set(

                key,

                value
            );
        },

        has(key) {

            return cache.has(
                key
            );
        },

        size() {

            return cache.size;
        }
    };
}

// Independent caches

const cache1 =

    createCache();

const cache2 =

    createCache();

cache1.set(

    "user",

    "Priya"
);

console.log(

    cache1.size()
);

console.log(

    cache2.size()
);

// Singleton Example

// cache.js

const sharedCache =

    createCache();

export default sharedCache;

// Any module importing
// sharedCache receives
// the same instance.

Output:
1
0

Bonus Reactive Counter with Proxy

Create a reactive object using Proxy. 

function reactive(

    obj,

    onChange
) {

    return new Proxy(

        obj,

        {

            set(

                target,

                prop,

                value
            ) {

                target[prop] =

                    value;

                onChange(

                    prop,

                    value
                );

                return true;
            }
        }
    );
}

const state =

    reactive(

        {

            count: 0
        },

        (prop, value) => {

            console.log(

                `${prop} → ${value}`
            );
        }
    );

state.count = 1;

state.count = 5;

state.username =

    "Priya";

Output:
count → 1
count → 5
username → Priya
